document.addEventListener('DOMContentLoaded', () => { const vehicleList = document.getElementById('vehicle-list'); const form = document.getElementById('vehicle-form'); const discordID = localStorage.getItem('userDiscordID'); if (!discordID) { vehicleList.innerHTML = `

Discord ID not found. Please log in or set it in localStorage.

`; form.style.display = 'none'; return; } // Load vehicles fetch('https://ckrp-backend.onrender.com/vehicles') .then(res => res.json()) .then(data => { vehicleList.innerHTML = ''; if (data.length === 0) { vehicleList.innerHTML = '

No vehicles listed yet.

'; return; } data.forEach(vehicle => { const div = document.createElement('div'); div.className = 'vehicle'; div.innerHTML = `

${vehicle.name}

Miles: ${vehicle.miles}

Condition: ${vehicle.condition}

Added by: ${vehicle.added_by}

${vehicle.image ? `` : ''} `; vehicleList.appendChild(div); }); }) .catch(err => { console.error('Error loading vehicles:', err); vehicleList.innerHTML = '

Failed to load vehicles.

'; }); // Submit form form.addEventListener('submit', async (e) => { e.preventDefault(); const vehicle = { name: document.getElementById('name').value.trim(), miles: parseInt(document.getElementById('miles').value), condition: document.getElementById('condition').value.trim(), image: document.getElementById('image').value.trim(), added_by: discordID }; console.log('Submitting vehicle:', vehicle); try { const res = await fetch('https://ckrp-backend.onrender.com/vehicles', { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(vehicle) }); const result = await res.json(); if (res.ok) { alert('Vehicle added successfully!'); form.reset(); location.reload(); // Refresh to show new vehicle } else { alert(`Error: ${result.detail || 'Failed to add vehicle.'}`); } } catch (err) { console.error('Submission error:', err); alert('Failed to submit vehicle. Check console for details.'); } }); });
